build:frontend:
  stage: build
  script:
    - cd frontend
    - pwd
    - npm run build

  
build:session:
  stage: build
  script:
    - cd server-session
    - pwd
    - gradle build

build:game:
  stage: build
  script:
    - cd server-game
    - pwd
    - gradle build

test:session:
  stage: test
  needs: ["build:session"]
  script:
    - gradle test

test:game:
  stage: test
  needs: ["build:game"]
  script:
    - gradle test

deploy-prod:
  stage: deploy
  needs: ["build:frontend", "test:session", "test:game"]
  script:
    - echo "This job deploys something from the $CI_COMMIT_BRANCH branch."
  environment: production
